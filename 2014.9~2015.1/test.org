#include<stdio.h>
#include<string.h>
#pragma warning(disable:4996)

int num[100000],sum[100000];

int main()
{
	int T;
	int m;
	while (scanf("%d", &T)==1)
	{
		for (int i = 0; i < T; i++)
			scanf("%d", &num[i]);
		for (int j = 0; j < T; j++)
		{
			m = num[j] % 10;
			if (m == 0||m==1||m==5||m==6)
			{
				sum[j] = m;
			}
			else if (m == 2)
			{
				switch (num[j] % 4 )
				{
				case 1:sum[j] = 2; break;
				case 2:sum[j] = 4; break;
				case 3:sum[j] = 8; break;
				case 0:sum[j] = 6; break;
				}
			}
			else if (m == 3)
			{
				switch (num[j] % 4)
				{
				case 1:sum[j] = 3; break;
				case 2:sum[j] = 9; break;
				case 3:sum[j] = 7; break;
				case 0:sum[j] = 1; break;
				}
			}
			else if (m == 4)
			{
				switch (num[j] % 2 )
				{
				case 1:sum[j] = 4; break;
				case 0:sum[j] = 6; break;
				}
			}
			else if (m == 7)
			{
				switch (num[j] % 4 )
				{
				case 1:sum[j] = 7; break;
				case 2:sum[j] = 9; break;
				case 3:sum[j] = 3; break;
				case 0:sum[j] = 1; break;
				}
			}
			else if (m == 8)
			{
				switch (num[j] % 4)
				{
				case 1:sum[j] = 8; break;
				case 2:sum[j] = 4; break;
				case 3:sum[j] = 2; break;
				case 0:sum[j] = 6; break;
				}
			}
			else if (m == 9)
			{
				switch (num[j] % 2 )
				{
				case 1:sum[j] = 9; break;
				case 0:sum[j] = 1; break;
				}
			}
		}
		for (int i = 0; i < T;i++)
			printf("%d\n", sum[i]);
	}
	return 0;
}
